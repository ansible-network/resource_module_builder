# Using replaced

- name: Configure following interfaces and replace their existing config
  ios_interfaces:
    config:
      - name: GigabitEthernet0/2
        description: Configured by Ansible
        enabled: True
	mtu: 2000
      - name: GigabitEthernet0/3
        description: 'Configured by Ansible Network'
        enabled: False
	duplex: auto
    operation: replaced
