
# Using overridden

- name: Override interfaces
  ios_interfaces:
    config:
      - name: GigabitEthernet0/2
        description: 'Configured by Ansible'
        enabled: True
        duplex: auto
      - name: GigabitEthernet0/3
        description: 'Configured by Ansible Network'
        enabled: False
        speed: 1000
    operation: overridden
